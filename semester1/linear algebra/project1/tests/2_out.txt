1. There are 15 ways in which the set can be partitioned.

2. The partitions are the following: 

{a1, a2, a3, a4} -> A × A
{ {a1, a2, a3}, {a4} } -> dA U {(a1, a2), (a1, a3), (a2, a1), (a2, a3), (a3, a1), (a3, a2)}
{ {a1, a2, a4}, {a3} } -> dA U {(a1, a2), (a1, a4), (a2, a1), (a2, a4), (a4, a1), (a4, a2)}
{ {a1, a2}, {a3, a4} } -> dA U {(a1, a2), (a2, a1), (a3, a4), (a4, a3)}
{ {a1, a2}, {a3}, {a4} } -> dA U {(a1, a2), (a2, a1)}
{ {a1, a3, a4}, {a2} } -> dA U {(a1, a3), (a1, a4), (a3, a1), (a3, a4), (a4, a1), (a4, a3)}
{ {a1, a3}, {a2, a4} } -> dA U {(a1, a3), (a3, a1), (a2, a4), (a4, a2)}
{ {a1, a3}, {a2}, {a4} } -> dA U {(a1, a3), (a3, a1)}
{ {a1, a4}, {a2, a3} } -> dA U {(a1, a4), (a4, a1), (a2, a3), (a3, a2)}
{ {a1}, {a2, a3, a4} } -> dA U {(a2, a3), (a2, a4), (a3, a2), (a3, a4), (a4, a2), (a4, a3)}
{ {a1}, {a2, a3}, {a4} } -> dA U {(a2, a3), (a3, a2)}
{ {a1, a4}, {a2}, {a3} } -> dA U {(a1, a4), (a4, a1)}
{ {a1}, {a2, a4}, {a3} } -> dA U {(a2, a4), (a4, a2)}
{ {a1}, {a2}, {a3, a4} } -> dA U {(a3, a4), (a4, a3)}
{ {a1}, {a2}, {a3}, {a4} } -> dA
